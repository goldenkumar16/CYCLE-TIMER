<script>
    let masterLogs = [];
    let state = {
        QC: { mode: 'LOAD', startTime: null, lastTime: null, interval: null, isPaused: false, mudaStart: null, totalMuda: 0, currentCycle: {} },
        RTG: { mode: 'LOAD', startTime: null, lastTime: null, interval: null, isPaused: false, mudaStart: null, totalMuda: 0, currentCycle: {} }
    };

    const config = {
        QC: {
            LOAD: ["Park to Truck", "Truck to Lock", "Lock to Sill", "Sill to Vessel", "Vessel to Unlock", "Unlock to Park"],
            DISCHARGE: ["Park to Vessel", "Vessel to Lock", "Lock to Sill", "Sill to Truck", "Truck to Unlock", "Unlock to Park"]
        },
        RTG: {
            LOAD: ["Standby to Lock (Bay)", "Lock to Standby", "Standby to Unlock (Trk)", "Unlock to Standby"],
            DISCHARGE: ["Park to Lock (Bay)", "Lock (Trk) to Unlock (Bay)", "Unlock (Bay) to Trk Line", "Trk Line to Park"]
        }
    };

    function showScreen(id) {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        if(id !== 'home-screen') render(id.split('-')[0].toUpperCase());
    }

    function toggleMuda(crane) {
        const s = state[crane];
        const btn = document.getElementById(crane.toLowerCase() + '-muda');
        if (!s.isPaused) {
            s.isPaused = true;
            s.mudaStart = Date.now();
            btn.classList.add('muda-active');
            btn.innerHTML = '<i class="fas fa-play"></i> RESUME FROM MUDA';
            clearInterval(s.interval);
        } else {
            s.isPaused = false;
            let waste = (Date.now() - s.mudaStart) / 1000;
            s.totalMuda += waste;
            // Shift the start time forward so the timer "stands still"
            s.startTime += (Date.now() - s.mudaStart);
            s.lastTime += (Date.now() - s.mudaStart);
            btn.classList.remove('muda-active');
            btn.innerHTML = '<i class="fas fa-pause"></i> PAUSE (MUDA)';
            startClock(crane);
        }
    }

    function startClock(crane) {
        const s = state[crane];
        s.interval = setInterval(() => {
            document.getElementById(crane.toLowerCase()+'-timer').innerText = ((Date.now() - s.startTime)/1000).toFixed(2);
        }, 50);
    }

    function triggerStep(crane, stepNum) {
        const now = Date.now();
        const s = state[crane];
        const steps = config[crane][s.mode];
        if (s.isPaused) return;

        if (stepNum === 1) {
            s.startTime = now;
            s.totalMuda = 0;
            s.currentCycle = { 
                Date: new Date().toLocaleDateString(),
                Equipment: crane,
                Unit_No: document.getElementById(crane.toLowerCase() + 'No').value || "N/A",
                Staff_ID: document.getElementById(crane.toLowerCase() + 'StaffId').value || "N/A",
                Type: s.mode
            };
            document.getElementById(crane.toLowerCase() + '-muda').disabled = false;
            startClock(crane);
        } else {
            s.currentCycle['Step_'+(stepNum-1)] = ((now - s.lastTime) / 1000).toFixed(2);
        }

        if (stepNum === steps.length) {
            clearInterval(s.interval);
            const total = ((now - s.startTime) / 1000).toFixed(2);
            s.currentCycle['MUDA_Time'] = s.totalMuda.toFixed(2);
            s.currentCycle['Net_Cycle_Time'] = (total - s.totalMuda).toFixed(2);
            s.currentCycle['Gross_Total'] = total;
            masterLogs.push({...s.currentCycle});
            document.getElementById(crane.toLowerCase() + '-muda').disabled = true;
            setTimeout(() => resetCrane(crane), 1200);
        } else {
            updateStepButtons(crane, stepNum, steps.length);
        }
        s.lastTime = now;
    }

    function updateStepButtons(crane, stepNum, total) {
        for(let i=1; i<=total; i++) document.getElementById(`${crane}-s${i}`).disabled = true;
        const next = document.getElementById(`${crane}-s${stepNum+1}`);
        next.disabled = false;
        next.style.background = crane === 'QC' ? 'var(--qc)' : 'var(--rtg)';
    }

    function resetCrane(crane) {
        state[crane].startTime = null;
        document.getElementById(crane.toLowerCase()+'-timer').innerText = "0.00";
        render(crane);
    }

    function setMode(crane, mode) {
        if(state[crane].startTime) return;
        state[crane].mode = mode;
        render(crane);
    }

    function render(crane) {
        const container = document.getElementById(crane.toLowerCase() + '-btns');
        container.innerHTML = '';
        config[crane][state[crane].mode].forEach((text, i) => {
            const btn = document.createElement('button');
            btn.className = 'btn-step' + (i === config[crane][state[crane].mode].length - 1 ? ' finish' : '');
            btn.disabled = i !== 0;
            btn.id = `${crane}-s${i+1}`;
            btn.innerText = (i+1) + ". " + text;
            btn.onclick = () => triggerStep(crane, i + 1);
            if(i === 0) btn.style.background = crane === 'QC' ? 'var(--qc)' : 'var(--rtg)';
            container.appendChild(btn);
        });
    }

    function downloadReport(type) {
        const filtered = masterLogs.filter(log => log.Equipment === type);
        if (filtered.length === 0) return alert(`No ${type} data.`);
        const headers = Object.keys(filtered[0]).join(",");
        let csv = "data:text/csv;charset=utf-8," + headers + "\n";
        filtered.forEach(row => csv += Object.values(row).join(",") + "\n");
        const link = document.createElement("a");
        link.href = encodeURI(csv);
        link.download = `${type}_Report_${new Date().toLocaleDateString()}.csv`;
        link.click();
    }
